17  package org.apache.commons.beanutils.bugs;
18  
19  import junit.framework.TestCase;
20  
21  import org.apache.commons.beanutils.AlphaBean;
22  import org.apache.commons.beanutils.BeanUtilsBean;
23  import org.apache.commons.beanutils.SuppressPropertiesBeanIntrospector;
24  
25  /**
26   * Class loader vulnerability in DefaultResolver
27   *
28   * @version $Id: Jira463TestCase.java 1632171 2014-10-15 20:15:17Z ggregory $
29   * @see <a href="https://issues.apache.org/jira/browse/BEANUTILS-463">https://issues.apache.org/jira/browse/BEANUTILS-463</a>
30   */
31  public class Jira463TestCase extends TestCase {
32      /**
33       * Tests that with a specialized {@code BeanIntrospector} implementation the class
34       * property can be suppressed.
35       */
36      public void testSuppressClassProperty() throws Exception {
37          final BeanUtilsBean bub = new BeanUtilsBean();
38          bub.getPropertyUtils().addBeanIntrospector(
39                  SuppressPropertiesBeanIntrospector.SUPPRESS_CLASS);
40          final AlphaBean bean = new AlphaBean();
41          try {
42              bub.getProperty(bean, "class");
43              fail("Could access class property!");
44          } catch (final NoSuchMethodException ex) {
45              // ok
46          }
47      }
48  }
