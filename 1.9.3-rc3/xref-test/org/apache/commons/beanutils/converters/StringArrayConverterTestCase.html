18  package org.apache.commons.beanutils.converters;
19  
20  import junit.framework.TestCase;
21  
22  /**
23   * Test Case for StringArrayConverter
24   *
25   * @version $Id: StringArrayConverterTestCase.java 1632171 2014-10-15 20:15:17Z ggregory $
26   */
27  
28  public class StringArrayConverterTestCase extends TestCase {
29  
30      public StringArrayConverterTestCase(final String name) {
31          super(name);
32      }
33  
34      public void testIntToString() {
35  
36          final int[] testArray = {1, 2, 3, 4, 5};
37          final String[] results = (String []) new StringArrayConverter().convert(String.class, testArray);
38  
39          assertEquals("Incorrect results size", 5, results.length);
40          assertEquals("Entry one is wrong", "1", results[0]);
41          assertEquals("Entry two is wrong", "2", results[1]);
42          assertEquals("Entry three is wrong", "3", results[2]);
43          assertEquals("Entry four is wrong", "4", results[3]);
44          assertEquals("Entry five is wrong", "5", results[4]);
45      }
46  }
